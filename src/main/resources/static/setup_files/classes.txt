CREATE TABLE IF NOT EXISTS classes (id int(10) UNSIGNED NOT NULL AUTO_INCREMENT,code varchar(10) COLLATE utf8_unicode_ci NOT NULL,name varchar(255) COLLATE utf8_unicode_ci NOT NULL,categoryId int(10) COLLATE utf8_unicode_ci NOT NULL,created_at timestamp NULL DEFAULT current_timestamp(),updated_at timestamp NULL DEFAULT current_timestamp(), PRIMARY KEY (id)) ENGINE=MyISAM AUTO_INCREMENT=1 DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci;
